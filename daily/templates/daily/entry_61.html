This entry describes a simple example of setting up webkit with go.

Install Pre-requisites

sudo apt-get install golang webkit2gtk-4.0 webkit2gtk-4.0

Write Script

Create main.go

package main

import "github.com/webview/webview"

func main() {
	debug := true
	w := webview.New(debug)
	defer w.Destroy()
	w.SetTitle("Minimal webview example")
	w.SetSize(800, 600, webview.HintNone)
	w.Navigate("https://www.youtube.com/watch?v=kOdrRIMYQgI")
	w.Run()
}

Run Script

go run main.go
